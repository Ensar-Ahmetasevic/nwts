** Import Google Fonts in next.js 14 **
(https://prismic.io/blog/nextjs-fonts)
(https://nextjs.org/docs/pages/building-your-application/optimizing/fonts)


游뚴 IMPORT 

// src/app/layout.js

import './globals.css'

//游녢 Import our second font
import { Open_Sans, Roboto_Mono } from 'next/font/google'

const openSans = Open_Sans({
  subsets: ['latin'],
  display: 'swap',
  //游녢 Add variable to our object
  variable: '--font-opensans',
})

//游녢 Configure the object for our second font
const robotoMono = Roboto_Mono({
  subsets: ['latin'],
  display: 'swap',
  variable: '--font-roboto-mono',
})

export default function RootLayout({ children }) {
  return (
    // 游녢 Attach CSS variables to HTML
    <html
      lang="en"
      className={`${openSans.variable} ${robotoMono.variable}`}
    >
      <body>{children}</body>
    </html>
  )
}
